311  mib[0] = CTL_NET;

348 template<class ForwardIterator>
349 void Tins::Utils::route_entries(ForwardIterator output) {
350  std::vector<char> buffer = query_route_table();
351  char *next = &buffer[0], *end = &buffer[buffer.size()];
352  rt_msghdr *rtm;
353  std::vector<sockaddr*> sa(RTAX_MAX);
354  char iface_name[IF_NAMESIZE];
355  while(next < end) {
356  rtm = (rt_msghdr*)next;
357  parse_header(rtm, sa.begin());
358  if (sa[RTAX_DST] && sa[RTAX_GATEWAY] && if_indextoname(rtm->rtm_index, iface_name)) {
359  RouteEntry entry;
360  entry.destination = IPv4Address(((struct sockaddr_in *)sa[RTAX_DST])->sin_addr.s_addr);
361  entry.gateway = IPv4Address(((struct sockaddr_in *)sa[RTAX_GATEWAY])->sin_addr.s_addr);
362  if(sa[RTAX_GENMASK])
363  entry.mask = IPv4Address(((struct sockaddr_in *)sa[RTAX_GENMASK])->sin_addr.s_addr);
364  else
365  entry.mask = IPv4Address(uint32_t());
366  entry.interface = iface_name;
367  *output++ = entry;
368  }
369  next += rtm->rtm_msglen;
370  }
371 }

398 void Tins::Utils::route_entries(ForwardIterator output) {
399  using namespace Tins::Internals;
400  std::ifstream input("/proc/net/route");
401  std::string destination, mask, gw;
402  uint32_t dummy;
403  skip_line(input);
404  RouteEntry entry;
405  while(input >> entry.interface >> destination >> gw) {
406  for(unsigned i(0); i < 5; ++i)
407  input >> mask;
408  from_hex(destination, dummy);
409  entry.destination = IPv4Address(dummy);
410  from_hex(mask, dummy);
411  entry.mask = IPv4Address(dummy);
412  from_hex(gw, dummy);
413  entry.gateway = IPv4Address(dummy);
414  skip_line(input);
