841  template<typename InputIterator>
842  void class_option_data2option(InputIterator start, InputIterator end,
843  std::vector<uint8_t>& buffer, size_t start_index = 0)
844  {
845  size_t index = start_index;
846  while(start != end) {
847  buffer.resize(buffer.size() + sizeof(uint16_t) + start->size());
848  *(uint16_t*)&buffer[index] = Endian::host_to_be<uint16_t>(start->size());
849  index += sizeof(uint16_t);
850  std::copy(start->begin(), start->end(), buffer.begin() + index);
851  index += start->size();
852 
853  start++;
854  }
855  }

857  template<typename OutputType>
858  OutputType option2class_option_data(const uint8_t *ptr, uint32_t total_sz) const
859  {
860  typedef typename OutputType::value_type value_type;
861  OutputType output;
862  size_t index = 0;
863  while(index + 2 < total_sz) {
864  uint16_t size = Endian::be_to_host(
865  *(const uint16_t*)(ptr + index)
866  );
867  index += sizeof(uint16_t);
868  if(index + size > total_sz)
869  throw option_not_found();
870  output.push_back(
871  value_type(ptr + index, ptr + index + size)
872  );
873  index += size;
874  }
875  if(index != total_sz)
876  throw option_not_found();
877  return output;
878  }
