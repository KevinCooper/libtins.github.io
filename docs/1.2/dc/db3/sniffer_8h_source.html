397  template<class Functor>
398  void Tins::BaseSniffer::sniff_loop(Functor function, uint32_t max_packets) {
399  for(iterator it = begin(); it != end(); ++it) {
400  try {
401  // If the functor returns false, we're done
402  if(!function(*it))
403  return;
404  }
405  catch(malformed_packet&) { }
406  catch(pdu_not_found&) { }
407  if(max_packets && --max_packets == 0)
408  return;
409  }
410  }
