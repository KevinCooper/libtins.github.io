270  template<size_t n>
271  void chaddr(const HWAddress<n> &new_chaddr) {
272  // Copy the new addr
273  uint8_t *end = std::copy(
274  new_chaddr.begin(),
275  new_chaddr.begin() + std::min(n, sizeof(_bootp.chaddr)),
276  _bootp.chaddr
277  );
278  // Fill what's left with zeros
279  if(end < _bootp.chaddr + chaddr_type::address_size)
280  std::fill(end, _bootp.chaddr + chaddr_type::address_size, 0);
281  }
