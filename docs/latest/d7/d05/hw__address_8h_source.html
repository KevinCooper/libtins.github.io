352 template<size_t n, typename Storage>
353 template<typename OutputIterator>
354 void HWAddress<n, Storage>::convert(const std::string &hw_addr,
355  OutputIterator output)
356 {
357  unsigned i(0);
358  size_t count(0);
359  storage_type tmp;
360  while(i < hw_addr.size() && count < n) {
361  const unsigned end = i+2;
362  tmp = storage_type();
363  while(i < end) {
364  if(hw_addr[i] >= 'a' && hw_addr[i] <= 'f')
365  tmp = (tmp << 4) | (hw_addr[i] - 'a' + 10);
366  else if(hw_addr[i] >= 'A' && hw_addr[i] <= 'F')
367  tmp = (tmp << 4) | (hw_addr[i] - 'A' + 10);
368  else if(hw_addr[i] >= '0' && hw_addr[i] <= '9')
369  tmp = (tmp << 4) | (hw_addr[i] - '0');
370  else if(hw_addr[i] == ':')
371  break;
372  else
373  throw std::runtime_error("Invalid byte found");
374  i++;
375  }
376  *(output++) = tmp;
377  count++;
378  if(i < hw_addr.size()) {
379  if(hw_addr[i] == ':')
380  i++;
381  else
382  throw std::runtime_error("Invalid separator");
383  }
384  }
385  while(count++ < n) {
386  *(output++) = storage_type();
387  }
388 }
