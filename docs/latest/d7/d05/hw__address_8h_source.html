335 template<size_t n, typename Storage>
336 template<typename OutputIterator>
337 void HWAddress<n, Storage>::convert(const std::string &hw_addr,
338  OutputIterator output)
339 {
340  unsigned i(0);
341  size_t count(0);
342  storage_type tmp;
343  while(i < hw_addr.size() && count < n) {
344  const unsigned end = i+2;
345  tmp = storage_type();
346  while(i < end) {
347  if(hw_addr[i] >= 'a' && hw_addr[i] <= 'f')
348  tmp = (tmp << 4) | (hw_addr[i] - 'a' + 10);
349  else if(hw_addr[i] >= 'A' && hw_addr[i] <= 'F')
350  tmp = (tmp << 4) | (hw_addr[i] - 'A' + 10);
351  else if(hw_addr[i] >= '0' && hw_addr[i] <= '9')
352  tmp = (tmp << 4) | (hw_addr[i] - '0');
353  else if(hw_addr[i] == ':')
354  break;
355  else
356  throw std::runtime_error("Invalid byte found");
357  i++;
358  }
359  *(output++) = tmp;
360  count++;
361  if(i < hw_addr.size()) {
362  if(hw_addr[i] == ':')
363  i++;
364  else
365  throw std::runtime_error("Invalid separator");
366  }
367  }
368  while(count++ < n) {
369  *(output++) = storage_type();
370  }
371 }
