368 template<size_t n, typename Storage>
369 template<typename OutputIterator>
370 void HWAddress<n, Storage>::convert(const std::string& hw_addr,
371  OutputIterator output) {
372  unsigned i(0);
373  size_t count(0);
374  storage_type tmp;
375  while (i < hw_addr.size() && count < n) {
376  const unsigned end = i+2;
377  tmp = storage_type();
378  while (i < end) {
379  if (hw_addr[i] >= 'a' && hw_addr[i] <= 'f') {
380  tmp = (tmp << 4) | (hw_addr[i] - 'a' + 10);
381  }
382  else if (hw_addr[i] >= 'A' && hw_addr[i] <= 'F') {
383  tmp = (tmp << 4) | (hw_addr[i] - 'A' + 10);
384  }
385  else if (hw_addr[i] >= '0' && hw_addr[i] <= '9') {
386  tmp = (tmp << 4) | (hw_addr[i] - '0');
387  }
388  else if (hw_addr[i] == ':') {
389  break;
390  }
391  else {
392  throw std::runtime_error("Invalid byte found");
393  }
394  i++;
395  }
396  *(output++) = tmp;
397  count++;
398  if (i < hw_addr.size()) {
399  if (hw_addr[i] == ':') {
400  i++;
401  }
402  else {
403  throw std::runtime_error("Invalid separator");
404  }
405  }
406  }
407  while (count++ < n) {
408  *(output++) = storage_type();
409  }
410 }
