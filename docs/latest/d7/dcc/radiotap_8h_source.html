490  template <size_t n>
491  void align_buffer(const uint8_t* buffer_start, const uint8_t*& buffer, uint32_t& size) {
492  uint32_t offset = ((buffer - buffer_start) % n);
493  if(offset) {
494  offset = n - offset;
495  if (offset > size) {
496  throw malformed_packet();
497  }
498  buffer += offset;
499  size -= offset;
500  }
