899 template<typename InputIterator>
900 void class_option_data2option(InputIterator start, InputIterator end,
901  std::vector<uint8_t>& buffer, size_t start_index = 0)
902 {
903  size_t index = start_index;
904  uint16_t uint16_t_buffer;
905  while(start != end) {
906  buffer.resize(buffer.size() + sizeof(uint16_t) + start->size());
907  uint16_t_buffer = Endian::host_to_be(static_cast<uint16_t>(start->size()));
908  std::memcpy(&buffer[index], &uint16_t_buffer, sizeof(uint16_t));
909  index += sizeof(uint16_t);
910  std::copy(start->begin(), start->end(), buffer.begin() + index);
911  index += start->size();
912 
913  start++;
914  }
915 }

917 template<typename OutputType>
918 OutputType option2class_option_data(const uint8_t *ptr, uint32_t total_sz)
919 {
920  typedef typename OutputType::value_type value_type;
921  OutputType output;
922  size_t index = 0;
923  while(index + 2 < total_sz) {
924  uint16_t size;
925  std::memcpy(&size, ptr + index, sizeof(uint16_t));
926  size = Endian::be_to_host(size);
927  index += sizeof(uint16_t);
928  if(index + size > total_sz)
929  throw option_not_found();
930  output.push_back(
931  value_type(ptr + index, ptr + index + size)
932  );
933  index += size;
934  }
935  if(index != total_sz)
936  throw malformed_option();
937  return output;
938 }
939 }
