884 template<typename InputIterator>
885 void class_option_data2option(InputIterator start, InputIterator end,
886  std::vector<uint8_t>& buffer, size_t start_index = 0)
887 {
888  size_t index = start_index;
889  while(start != end) {
890  buffer.resize(buffer.size() + sizeof(uint16_t) + start->size());
891  *(uint16_t*)&buffer[index] = Endian::host_to_be<uint16_t>(start->size());
892  index += sizeof(uint16_t);
893  std::copy(start->begin(), start->end(), buffer.begin() + index);
894  index += start->size();
895 
896  start++;
897  }
898 }

900 template<typename OutputType>
901 OutputType option2class_option_data(const uint8_t *ptr, uint32_t total_sz)
902 {
903  typedef typename OutputType::value_type value_type;
904  OutputType output;
905  size_t index = 0;
906  while(index + 2 < total_sz) {
907  uint16_t size = Endian::be_to_host(
908  *(const uint16_t*)(ptr + index)
909  );
910  index += sizeof(uint16_t);
911  if(index + size > total_sz)
912  throw option_not_found();
913  output.push_back(
914  value_type(ptr + index, ptr + index + size)
915  );
916  index += size;
917  }
918  if(index != total_sz)
919  throw malformed_option();
920  return output;
921 }
