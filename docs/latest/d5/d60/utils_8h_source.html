355 template<class ForwardIterator>
356 void Tins::Utils::route_entries(ForwardIterator output) {
357  std::vector<char> buffer = query_route_table();
358  char *next = &buffer[0], *end = &buffer[buffer.size()];
359  rt_msghdr *rtm;
360  std::vector<sockaddr*> sa(RTAX_MAX);
361  char iface_name[IF_NAMESIZE];
362  while(next < end) {
363  rtm = (rt_msghdr*)next;
364  parse_header(rtm, sa.begin());
365  if (sa[RTAX_DST] && sa[RTAX_GATEWAY] && if_indextoname(rtm->rtm_index, iface_name)) {
366  RouteEntry entry;
367  entry.destination = IPv4Address(((struct sockaddr_in *)sa[RTAX_DST])->sin_addr.s_addr);
368  entry.gateway = IPv4Address(((struct sockaddr_in *)sa[RTAX_GATEWAY])->sin_addr.s_addr);
369  if(sa[RTAX_GENMASK])
370  entry.mask = IPv4Address(((struct sockaddr_in *)sa[RTAX_GENMASK])->sin_addr.s_addr);
371  else
372  entry.mask = IPv4Address(uint32_t());
373  entry.interface = iface_name;
374  *output++ = entry;
375  }
376  next += rtm->rtm_msglen;
377  }
