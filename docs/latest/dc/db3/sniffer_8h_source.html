580  template<class Functor>
581  void Tins::BaseSniffer::sniff_loop(Functor function, uint32_t max_packets) {
582  for(iterator it = begin(); it != end(); ++it) {
583  try {
584  // If the functor returns false, we're done
585  #if TINS_IS_CXX11 && !defined(_MSC_VER)
586  if (!Tins::Internals::invoke_loop_cb(function, *it))
587  return;
588  #else
589  if(!function(*it->pdu()))
590  return;
591  #endif
592  }
593  catch(malformed_packet&) { }
594  catch(pdu_not_found&) { }
595  if(max_packets && --max_packets == 0)
596  return;
597  }
598  }
599 }
