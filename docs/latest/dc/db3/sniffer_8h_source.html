567  template<class Functor>
568  void Tins::BaseSniffer::sniff_loop(Functor function, uint32_t max_packets) {
569  for(iterator it = begin(); it != end(); ++it) {
570  try {
571  // If the functor returns false, we're done
572  if(!function(*it))
573  return;
574  }
575  catch(malformed_packet&) { }
576  catch(pdu_not_found&) { }
577  if(max_packets && --max_packets == 0)
578  return;
579  }
580  }
581 }
