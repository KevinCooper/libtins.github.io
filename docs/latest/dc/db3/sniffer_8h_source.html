420  template<class Functor>
421  void Tins::BaseSniffer::sniff_loop(Functor function, uint32_t max_packets) {
422  for(iterator it = begin(); it != end(); ++it) {
423  try {
424  // If the functor returns false, we're done
425  if(!function(*it))
426  return;
427  }
428  catch(malformed_packet&) { }
429  catch(pdu_not_found&) { }
430  if(max_packets && --max_packets == 0)
431  return;
432  }
433  }
