191  typename enable_if<
192  is_unsigned_integral<T>::value && is_unsigned_integral<U>::value
193  >::type
194  > {
195  template<typename X, typename PDUType>
196  static std::vector<std::pair<T, U> > convert(const PDUOption<X, PDUType>& opt) {
197  if(opt.data_size() % (sizeof(T) + sizeof(U)) != 0)
198  throw malformed_option();
199  const uint8_t *ptr = opt.data_ptr(), *end = ptr + opt.data_size();
200 
201  std::vector<std::pair<T, U> > output;
202  while(ptr < end) {
203  std::pair<T, U> data;
204  data.first = *(const T*)ptr;
205  ptr += sizeof(T);
206  data.second = *(const U*)ptr;
207  ptr += sizeof(U);
208  if(PDUType::endianness == PDUType::BE) {
209  data.first = Endian::be_to_host(data.first);
210  data.second = Endian::be_to_host(data.second);
211  }
212  else {
213  data.first = Endian::le_to_host(data.first);
214  data.second = Endian::le_to_host(data.second);
215  }
216  output.push_back(data);
217  }

494  template<typename ForwardIterator>
495  void set_payload_contents(ForwardIterator start, ForwardIterator end) {
496  size_t total_size = std::distance(start, end);
497  if (total_size > std::numeric_limits<uint16_t>::max()) {
498  throw option_payload_too_large();
499  }
500  real_size_ = static_cast<uint16_t>(total_size);
501  if(real_size_ <= small_buffer_size) {
502  std::copy(
503  start,
504  end,
505  payload_.small_buffer
506  );
507  }
508  else {
509  payload_.big_buffer_ptr = new data_type[real_size_];
510  uint8_t* ptr = payload_.big_buffer_ptr;
511  while (start < end) {
512  *ptr = *start;
513  ++ptr;
514  ++start;
515  }
516  }
517  }
