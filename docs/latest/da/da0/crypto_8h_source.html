511 template<typename ForwardIterator>
512 RC4Key::RC4Key(ForwardIterator start, ForwardIterator end) {
513  for (size_t i = 0; i < data_size; ++i) {
514  data[i] = static_cast<uint8_t>(i);
515  }
516  size_t j = 0;
517  ForwardIterator iter = start;
518  for (size_t i = 0; i < data_size; ++i) {
519  j = (j + data[i] + *iter++) % 256;
520  if(iter == end) {
521  iter = start;
522  }
523  std::swap(data[i], data[j]);
524  }
525 }

527 template<typename ForwardIterator, typename OutputIterator>
528 void rc4(ForwardIterator start, ForwardIterator end, RC4Key& key, OutputIterator output) {
529  size_t i = 0, j = 0;
530  while (start != end) {
531  i = (i + 1) % RC4Key::data_size;
532  j = (j + key.data[i]) % RC4Key::data_size;
533  std::swap(key.data[i], key.data[j]);
534  *output++ = *start++ ^ key.data[(key.data[i] + key.data[j]) % RC4Key::data_size];
535  }
536 }
