346 template<typename DataFunctor, typename EndFunctor>
347 bool TCPStreamFollower::callback(PDU &pdu, const DataFunctor &data_fun, const EndFunctor &end_fun) {
348  IP &ip = pdu.rfind_pdu<IP>();
349  TCP &tcp = pdu.rfind_pdu<TCP>();
350  TCPStream::StreamInfo info(
351  ip.src_addr(), ip.dst_addr(),
352  tcp.sport(), tcp.dport()
353  );
354  sessions_type::iterator it = sessions.find(info);
355  if(it == sessions.end()) {
356  std::swap(info.client_addr, info.server_addr);
357  std::swap(info.client_port, info.server_port);
358  if((it = sessions.find(info)) == sessions.end()) {
359  if(tcp.get_flag(TCP::SYN) && !tcp.get_flag(TCP::ACK)) {
360  sessions.insert(
361  std::make_pair(
362  info,
363  TCPStream(&ip, &tcp, last_identifier++)
364  )
365  );
366  }
367  return true;
368  }
369  }
370  if(it->second.update(&ip, &tcp))
371  data_fun(it->second);
372  // We're done with this stream
373  if(it->second.is_finished()) {
374  end_fun(it->second);
375  sessions.erase(it);
376  }
377  return true;
378 }
